	def Value(self) -> int:
		"""
		Read-only property to return the internal value.

		:return: Internal value.
		"""
		return self._value

	def Add(self, value) -> None:
		"""
		Accumulate value to internal value.

		:param value: Value to accumulate.
		"""
		self._value += value
